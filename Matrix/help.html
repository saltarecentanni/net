#!/bin/bash
# /etc/cron.daily/matrix-backup.sh

BACKUP_DIR="/backups/matrix"
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=90

mkdir -p "$BACKUP_DIR"

# Download da API

# Verifica file non vuoto
if [ -s "$BACKUP_DIR/matrix_$DATE.json" ]; then
    echo "Backup OK: matrix_$DATE.json"
    
    # Cleanup vecchi backup
    find "$BACKUP_DIR" -name "matrix_*.json" -mtime +$RETENTION_DAYS -delete
else
    echo "ERRORE: Backup vuoto o fallito!"
    rm -f "$BACKUP_DIR/matrix_$DATE.json"
    exit 1
fi

  "connections": [...],
  "rooms": [...]

GET /api/devices

Descrizione: Lista solo dispositivi

Query params: ?location=Milan%20HQ&type=switch&status=active

GET /api/connections

Descrizione: Lista solo connessioni

Query params: ?status=active&type=fiber

GET /api/logs

Descrizione: Activity logs

Query params: ?limit=100&offset=0&user=admin

POST /api/validate

Descrizione: Valida JSON prima di import

Request: Multipart form-data con file JSON
